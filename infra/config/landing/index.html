Quick start
# 1 — clone and start
git clone <repo-url>
cd order-stack
cp infra/.env.example infra/.env
docker compose -f infra/compose.yml up -d

# 2 — wait ~30 s for health checks

# 3 — smoke test (via Traefik)
cd test && k6 run smoke-test-traefik.js

# 4 — full load test (~5k orders/min)
cd test && k6 run order-load-test.js
